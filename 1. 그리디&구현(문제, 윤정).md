[백준 5585 거스름돈](https://www.acmicpc.net/problem/5585)
```cpp
#include <iostream>
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int arr[] = {500, 100, 50, 10, 5, 1};
    int n, cnt=0;
    cin>>n;
    n = 1000-n;
    for(int i=0;i<6;i++){
        cnt += n/arr[i];
        n%=arr[i];
    }
    cout<<cnt<<'\n';
    return 0;
}
```
   
[백준 2541 잃어버린 괄호](https://www.acmicpc.net/problem/1541)
```cpp
#include <iostream>
#include <string>
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    string str, tmp="";
    int ans=0, flag = 0;
    cin>>str;
    for (int i = 0; i <= str.size(); i++){
        if (str[i] == '+' || str[i] == '-' || str[i] == '\0'){
            if (flag==1) ans -= stoi(tmp);
            else ans += stoi(tmp);
            tmp = ""; 
            //-가 하나라도 나오면 그 뒤 연산자는 모두 -임
            if (str[i] == '-') flag = 1;
            continue;
        }
        tmp += str[i];
    }
    cout<<ans;
}
```
   
[백준 2217 로프](https://www.acmicpc.net/problem/2217)
```cpp
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main() {
    int n, sum, max=0;
    cin>>n;
    vector<int> arr(n);
    for(int i=0;i<n;i++) cin>>arr[i];
    sort(arr.begin(), arr.end());
    for(int i =0; i<n; i++){
        sum = arr[i] * (n-i);
        if(sum > max)
            max = sum;
    }
    cout<<max;
}
```
# 브루트포스(완전탐색)
[백준 14501 퇴사](https://www.acmicpc.net/problem/14501)
```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int n,ans;
int t[16], p[16];

void dfs(int i, int sum) {
    if (i == n) {
        ans = max(ans, sum);
        return;
    }
    if (i + t[i] < n + 1)
        dfs(i + t[i], sum + p[i]);
    if (i < n)
        dfs(i + 1, sum);
}

int main() {
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> t[i] >> p[i];
    dfs(0,0);
    cout << ans;
}
```
