```cpp
#include <iostream>
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int arr[] = {500, 100, 50, 10, 5, 1};
    int n, cnt=0;
    cin>>n;
    n = 1000-n;
    for(int i=0;i<6;i++){

        n%=arr[i];
```

[백준 2541 잃어버린 괄호](https://www.acmicpc.net/problem/1541)
```cpp
#include <iostream>
#include <string>
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    string str, tmp="";
    int ans=0, flag = 0;
    cin>>str;
    for (int i = 0; i <= str.size(); i++){
        if (str[i] == '+' || str[i] == '-' || str[i] == '\0'){
            if (flag==1) ans -= stoi(tmp);
            else ans += stoi(tmp);
            tmp = ""; 
            //-가 하나라도 나오면 그 뒤 연산자는 모두 -임
            if (str[i] == '-') flag = 1;
            continue;
        }
        tmp += str[i];
    }
    cout<<ans;
}
```

[백준 2217 로프](https://www.acmicpc.net/problem/2217)
```cpp
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main() {
    int n, sum, max=0;
    cin>>n;
    vector<int> arr(n);
    for(int i=0;i<n;i++) cin>>arr[i];
    sort(arr.begin(), arr.end());
    for(int i =0; i<n; i++){
        sum = arr[i] * (n-i);
        if(sum > max)
            max = sum;
    }
    cout<<max;
}
```

[백준 11000 강의실 배정](https://www.acmicpc.net/problem/11000)
```cpp
#include<iostream>
#include<algorithm>
#include<vector>
#include <queue>
using namespace std;

int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0), cout.tie(0);
	int n, a, b;
	cin >> n;
	vector<pair<int, int>>v;
	priority_queue<int, vector<int>, greater<int> > pq;
	for (int i = 0; i < n; i++) {
		cin >> a >> b;
		v.push_back({ a,b});
	}
	sort(v.begin(), v.end());
	pq.push(v[0].second);

	for (int i = 1; i < n; ++i) {
		if (pq.top() <= v[i].first) {
			pq.pop();
			pq.push(v[i].second);
		}
		else {
			pq.push(v[i].second);
		}
	}
	cout << pq.size() << endl;
	return 0;
}
```
# 브루트포스(완전탐색)
[백준 14501 퇴사](https://www.acmicpc.net/problem/14501)
```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int n,ans;
int t[16], p[16];

void dfs(int i, int sum) {
    if (i == n) {
        ans = max(ans, sum);
        return;
    }
    if (i + t[i] < n + 1)
        dfs(i + t[i], sum + p[i]);
    if (i < n)
        dfs(i + 1, sum);
}

int main() {
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> t[i] >> p[i];
    dfs(0,0);
    cout << ans;
}
```
   
# 구현
[백준 10773 제로](https://www.acmicpc.net/problem/10773)
```cpp
#include <iostream>
#include <stack>
#include <algorithm>
using namespace std;

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int n, x, sum = 0;
	cin >> n;
	stack<int> st;
	for (int i = 0; i < n; i++) {
		cin >> x;
		if (x == 0) st.pop();
		else st.push(x);
	}
	while (!empty(st)) {
		sum += st.top();
		st.pop();
	}
	cout << sum;
	return 0;
}
```
